void minmax(int X[][8], int Y[][2], int Z[][2], int M, int N)
{
	int i, j, elem1 = 0, elem2 = 0, buf1 = 0, buf2 = 0, min, max, imin = 0, imax = 0;
	min = X[0][0];
	max = X[0][0];
	for (i=0; i<M; i++)
	{
		for (j=0; j<N; j++)
		{
			if (X[i][j] < min)
			{
				min = X[i][j];
				imin = i;
			}
			if (X[i][j] > max)
			{
				max = X[i][j];
				imax = i;
			}
		}
	}
	for (i=0; i<N; i++)
	{
		for (j=0; j<2; j++)
		{
			Y[i][0] = X[imin][i];
			Y[i][1] = X[imax][i];
			}
		}
	for (i=0; i<M; i++)
	{
		for (j=0; j<N; j++)
		{
			buf1 = X[i][j] * Y[j][0];
			buf2 = X[i][j] * Y[j][1];
			elem1 = elem1 + buf1;
			elem2 = elem2 + buf2;
		}
		Z[i][0] = elem1;
		Z[i][1] = elem2;
		elem1 = 0;
		elem2 = 0;
	}
}
