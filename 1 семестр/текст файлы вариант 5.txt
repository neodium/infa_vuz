void repeatCheck(FILE * f){
	setlocale(LC_ALL, "rus");
	int i, j, l = 0, k = 0, strCount = 1, strLength, uniqOcc[1000];
	char string[1000] = {0};
	char uniqC[1000] = {0};
	while(!feof(f)){
		printf("Уникальные символы строки #%d и их количество:\n", strCount);
		fgets(string, 255, f);
		for (i=0; i<strlen(string); i++){
			if (strchr(uniqC, string[i]) == 0 && string[i] != ' ' && string[i] != '\0' && string[i] != '\n'){
				uniqC[l] = string[i];
				l++;
			}
		}
		l = 0;
		for (i=0; i<strlen(uniqC); i++){
			for (j=0; j<strlen(string); j++){
				if (uniqC[i] == string[j]) k++;
			}
			uniqOcc[i] = k;
			printf("Символ: %c - %d\n", uniqC[i], uniqOcc[i]);
			k=0;
		}
		memset(uniqC, 0, sizeof(uniqC)); //очищает массив с уникальными символами в строке
		memset(uniqOcc, 0, sizeof(uniqOcc));//очищает массив с кол-вом уникальных символов в строке
		strCount+=1;
	}
}
